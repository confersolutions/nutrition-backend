Core Tables

recipes
CREATE TABLE recipes ( id UUID PRIMARY KEY, title TEXT NOT NULL, description TEXT, calories INTEGER, protein_g NUMERIC(5,2), diet_tags TEXT[], cuisines TEXT[], allergens TEXT[], tsv tsvector, search_text TEXT, status TEXT DEFAULT 'draft', market_country TEXT DEFAULT 'US' );
saved_recipes (RLS Enabled)
CREATE TABLE saved_recipes ( user_id TEXT, recipe_id UUID, saved_at TIMESTAMPTZ DEFAULT NOW(), PRIMARY KEY (user_id, recipe_id) ); -- Row Level Security Policy CREATE POLICY saved_recipes_owner_only ON saved_recipes USING (user_id = current_setting('app.current_user_id'));

Performance Features

Full-Text Search Indexes
-- GIN indexes for optimal search performance CREATE INDEX CONCURRENTLY idx_recipes_tsv ON recipes USING GIN (tsv); CREATE INDEX CONCURRENTLY idx_recipes_cuisines ON recipes USING GIN (cuisines); CREATE INDEX CONCURRENTLY idx_recipes_diet_tags ON recipes USING GIN (diet_tags);
Materialized Views
-- Auto-refreshing popularity metrics CREATE MATERIALIZED VIEW mv_recipe_popularity_30d AS SELECT recipe_id, COUNT(*) as cooked_30d FROM recipe_history WHERE event = 'cooked' AND at > NOW() - INTERVAL '30 days' GROUP BY recipe_id; -- Auto-refresh every 5 minutes SELECT cron.schedule( 'refresh-popularity-mv', '*/5 * * * *', 'REFRESH MATERIALIZED VIEW CONCURRENTLY mv_recipe_popularity_30d;' );

// Middleware for JWT verification async function verifyAppwriteJWT(request) { const jwt = request.headers['x-appwrite-jwt']; if (!jwt) { throw new Error('X-Appwrite-JWT header required'); } // Verify JWT with Appwrite const decoded = await appwrite.account.get(jwt); // Check admin privileges const isAdmin = decoded.profile?.role === 'admin' || await isTeamMember(decoded.$id, ADMINS_TEAM_ID); return { userId: decoded.$id, isAdmin }; }
